    const sendBtn = document.getElementById("send-btn");

    // Check server-side AI status (safe: does not reveal the key)
    const statusBadge = document.getElementById('ai-badge');
    if (statusBadge) {
      fetch('ask-ai.php').then(r => r.json()).then(d => {
        if (d.gemini && d.live) {
          statusBadge.textContent = 'Gemini: Enabled (live) ✅';
          statusBadge.style.color = 'green';
        } else if (d.gemini && !d.live) {
          statusBadge.textContent = 'Gemini: Configured but not live ⚠️ — check billing or API key';
          statusBadge.style.color = 'maroon';
        } else {
          statusBadge.textContent = 'Gemini: Disabled ⚠️ — create a .env file with GEMINI_API_KEY or set the environment variable';
          statusBadge.style.color = 'maroon';
        }
      }).catch(() => {
        statusBadge.textContent = 'Gemini: Status unknown (could not reach server)';
        statusBadge.style.color = '#666';
      });
    }

    function addMsg(text, cls, id) {
      const div = document.createElement("div");
      div.className = "msg " + cls;
      if (id) div.id = id;
      div.textContent = text;
      messages.appendChild(div);
      messages.scrollTop = messages.scrollHeight;
    }

    async function sendMessage() {
      const text = input.value.trim();
      if (!text) return;
      addMsg("You: " + text, "you");
      input.value = "";
      sendBtn.disabled = true;
      sendBtn.textContent = "Sending...";

        const placeholderId = "placeholder";
      addMsg("AI: ", "ai", placeholderId);
      const ph = document.getElementById(placeholderId);

      try {
        // Include live_only and model selection in the request body
        const liveOnly = document.getElementById('live-only') ? document.getElementById('live-only').checked : false;
        const model = document.getElementById('model-select') ? document.getElementById('model-select').value : undefined;
        const response = await fetch("ask-ai.php?stream=1", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ question: text, live_only: liveOnly, model: model })
        });

        if (!response.ok) {
          // If stream endpoint returns an error status, read text and show
          const txt = await response.text();
          if (ph) ph.remove();
          addMsg("AI Error: " + (txt || response.statusText), "error");
        } else if (!response.body) {
          // Fallback if streaming not available
          const data = await response.json();
          if (ph) ph.remove();
          if (data.error) {
            addMsg("AI Error: " + data.error, "error");
            if (statusBadge) { statusBadge.textContent = 'Gemini: Live error — using fallback'; statusBadge.style.color = 'maroon'; }
          } else {
            addMsg("AI: " + data.answer, "ai");
            if (data.note) {
              addMsg("Note: " + data.note, "note");
              const noteLower = data.note.toLowerCase();
              if (noteLower.includes('google') || noteLower.includes('gemini') || noteLower.includes('api error') || noteLower.includes('error')) {
                if (statusBadge) { statusBadge.textContent = 'Gemini: Live error — using fallback'; statusBadge.style.color = 'maroon'; }
              }
            }
          }
        } else {
          // Stream reader
          const reader = response.body.getReader();
          const decoder = new TextDecoder();
          let buffer = '';
          let done = false;

          while (!done) {
            const { value, done: readerDone } = await reader.read();
            if (value) {
              buffer += decoder.decode(value, { stream: true });
              const parts = buffer.split('\n\n');
              buffer = parts.pop(); // keep incomplete chunk
              for (const part of parts) {
                if (!part.trim()) continue;
                const cleaned = part.replace(/^data: /, '').trim();
                if (cleaned === '[DONE]') { done = true; break; }
                try {
                  const payload = JSON.parse(cleaned);
                  // Google simulated stream uses { text: '...' }
                  if (payload.text) {
                    ph.textContent += payload.text;
                    messages.scrollTop = messages.scrollHeight;
                    continue;
                  }
                  const delta = payload.choices && payload.choices[0] && payload.choices[0].delta && payload.choices[0].delta.content;
                  const content = payload.choices && payload.choices[0] && payload.choices[0].message && payload.choices[0].message.content;
                  if (delta) {
                    ph.textContent += delta;
                    messages.scrollTop = messages.scrollHeight;
                  }
                  if (content) {
                    ph.textContent += content;
                    messages.scrollTop = messages.scrollHeight;
                  }
                } catch (e) {
                  // ignore parse errors
                }
              }
            }
            if (readerDone) break;
          }

          // Process any remaining buffer
          if (buffer) {
            const lines = buffer.split('\n');
            for (const l of lines) {
              const cleaned = l.replace(/^data: /, '').trim();
              if (!cleaned || cleaned === '[DONE]') continue;
              try {
                const payload = JSON.parse(cleaned);
                const delta = payload.choices && payload.choices[0] && payload.choices[0].delta && payload.choices[0].delta.content;
                if (delta) ph.textContent += delta;
              } catch (e) {}
            }
          }
        }
      } catch (err) {
        const ph2 = document.getElementById(placeholderId);
        if (ph2) ph2.remove();
        addMsg("AI: Sorry, I couldn't connect. " + err.message, "error");
      } finally {
        sendBtn.disabled = false;
        sendBtn.textContent = "Send";
        input.focus();
      }
    }

    sendBtn.addEventListener("click", sendMessage);
    input.addEventListener("keydown", (e) => {
      if (e.key === "Enter") sendMessage();
    });
